前言

tcp是现在通信的基础，在http3没有到来之前这句话还是很对的。 因为tcp是流式传输，客户端一次write　1600个字节，服务端也许只读了1540个字节。 这就是著名的粘包问题，要解决这个问题也很简单，client 和 server约定下解析方式。 比如客户端发10个字节，可以把10这个信息也一起发过来，先发4个字节保存长度，再发数据就行。 server也按这个顺序解析。

use std::io::{Read, Write};
use std::net::TcpStream;
use structopt::StructOpt;

#[derive(StructOpt, Debug)]
struct Client {
    /// tcp server address
    #[structopt(short, long)]
    addr: String,

    /// data
    #[structopt(short, long)]
    data: String,
}

fn main() {
    // 解析命令行
    let c = Client::from_args();

    // ============1.连接服务端
    let mut stream = TcpStream::connect(c.addr).unwrap();

    // 写入req数据至服务端
    // 存放头
    let mut buffer = (c.data.len() as u32).to_be_bytes();

    // 写入头
    stream.write_all(&buffer).unwrap();

    // 写入body
    stream.write_all(c.data.as_bytes()).unwrap();

    // ============2.读取rsp数据
    // 读取4个字节定长数据
    stream.read_exact(&mut buffer[..]).unwrap();

    // 计算buffer长度
    let size = u32::from_be_bytes(buffer);

    // 数据body
    let mut payload = vec![0; size as usize];

    // 读取数据body
    stream.read_exact(&mut payload[..]).unwrap();

    // 返回终端
    println!("{}", std::str::from_utf8(&payload).unwrap());
}

use std::io;
use std::io::{Read, Write};
use std::net::{TcpListener, TcpStream};
use structopt::StructOpt;

#[derive(StructOpt, Debug)]
struct Server {
    /// server address
    #[structopt(short, long)]
    server_addr: String,
}

fn work_conn(mut stream: TcpStream) -> io::Result<()> {
    let mut buffer = [0u8; 4];
    // ==========读取4个字节定长数据
    stream.read_exact(&mut buffer[..])?;

    // 计算buffer长度
    let size = u32::from_be_bytes(buffer);

    // 数据body
    let mut payload = vec![0; size as usize];

    // 读取数据body
    stream.read_exact(&mut payload[..])?;

    // 追加一些数据，然后返回
    payload.append(&mut "#Server read".as_bytes().to_vec());

    // ============写入rsp数据
    let size = (payload.len() as u32).to_be_bytes();
    stream.write_all(&size)?;
    stream.write_all(&payload[..])?;
    Ok(())
}

impl Server {
    fn main_loop(&self) {
        // 创建listen
        let listener = TcpListener::bind(&self.server_addr).unwrap();

        for stream in listener.incoming() {
            let stream = match stream {
                Ok(stream) => stream,
                Err(e) => {
                    /* connection failed */
                    println!("{:#?}", e);
                    continue;
                }
            };

            work_conn(stream).unwrap();
        }
    }
}

fn main() {
    let s = Server::from_args();

    s.main_loop();
}
